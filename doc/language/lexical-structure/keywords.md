# Keywords

```ebnf
keyword ::= hard-keyword |
            soft-keyword |
            reserved-keyword
```

```ebnf
hard-keyword ::= "and" |
                 "as" |
                 "assert" |
                 "break" |
                 "catch" |
                 "cond" |
                 "const" |
                 "defer" |
                 "else" |
                 "err" |
                 "ext" |
                 "false" |
                 "fn" |
                 "if" |
                 "in" |
                 "let" |
                 "match" |
                 "mod" |
                 "mut" |
                 "next" |
                 "nil" |
                 "not" |
                 "opaque" |
                 "or" |
                 "pub" |
                 "raise" |
                 "rec" |
                 "recv" |
                 "ret" |
                 "tail" |
                 "test" |
                 "true" |
                 "type" |
                 "use" |
                 "while"
```

```ebnf
soft-keyword ::= "agent" |
                 "any" |
                 "atom" |
                 "bool" |
                 "handle" |
                 "int" |
                 "none" |
                 "real" |
                 "ref" |
                 "str"
```

```ebnf
reserved-keyword ::= "friend" |
                     "macro" |
                     "meta" |
                     "pragma" |
                     "quote" |
                     "try" |
                     "unquote" |
                     "yield"
```
